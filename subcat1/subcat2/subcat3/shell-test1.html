$@ And $*

test.sh

Output count of arguments, and all arguments

echo '$#  :' \"$#\"
cnt=$#

for (( i = 0; i < $cnt; i++ )); do
        echo "arg[$i]: $1"
        shift
done

test result

with double quote, the result is different.

~/wk$ set a b c
~/wk$ /tmp/test.sh "$@"
$#  : "3"
arg[0]: a
arg[1]: b
arg[2]: c
~/wk$ /tmp/test.sh "$*"
$#  : "1"
arg[0]: a b c
